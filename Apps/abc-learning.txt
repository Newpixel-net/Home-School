 <script>
 // ABC Learning Application
 class ABCLearning {
 constructor() {
 this.currentLetter = 'A';
 this.completedLetters = new Set();
 this.alphabet = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'.split('');

 this.init();
 }

 init() {
 console.log('&#128292; ABC Learning initializing...');

 // Generate alphabet grid
 this.generateAlphabetGrid();

 // Setup event listeners
 this.setupEventListeners();

 // Load saved progress
 this.loadProgress();

 console.log('&#9989; ABC Learning ready!');
 }

 generateAlphabetGrid() {
 const container = document.getElementById('letters-container');
 container.innerHTML = '';

 this.alphabet.forEach(letter => {
 const btn = document.createElement('button');
 btn.className = 'letter-btn';
 btn.textContent = letter;
 btn.dataset.letter = letter;

 if (letter === this.currentLetter) {
 btn.classList.add('active');
 }

 btn.addEventListener('click', () => {
 this.selectLetter(letter);
 });

 container.appendChild(btn);
 });
 }

 setupEventListeners() {
 document.getElementById('play-sound').addEventListener('click', () => {
 this.playLetterSound();
 });
 }

 selectLetter(letter) {
 console.log(`Letter selected: ${letter}`);
 this.currentLetter = letter;

 // Update display with animation
 const letterDisplay = document.getElementById('letter-large');
 letterDisplay.style.animation = 'none';

 setTimeout(() => {
 letterDisplay.textContent = letter;
 letterDisplay.style.animation = 'bounceIn 0.8s cubic-bezier(0.68, -0.55, 0.265, 1.55)';
 }, 10);

 // Update navigation
 document.querySelectorAll('.letter-btn').forEach(btn => {
 btn.classList.remove('active');
 if (btn.dataset.letter === letter) {
 btn.classList.add('active');
 }
 });

 // Auto-play sound
 this.playLetterSound();

 // Track with Firebase
 if (window.firebaseManager && window.firebaseManager.initialized) {
 window.firebaseManager.trackEvent('letter_selected', {
 letter: letter
 });
 }
 }

 playLetterSound() {
 const letter = this.currentLetter;

 // Use Web Speech API
 if ('speechSynthesis' in window) {
 // Cancel any ongoing speech
 window.speechSynthesis.cancel();

 // Create utterance
 const utterance = new SpeechSynthesisUtterance();
 utterance.text = `The letter ${letter}`;
 utterance.rate = 0.8;
 utterance.pitch = 1.2;
 utterance.volume = 1;

 // Speak
 window.speechSynthesis.speak(utterance);

 // Animate button
 const btn = document.getElementById('play-sound');
 btn.style.animation = 'none';
 setTimeout(() => {
 btn.style.animation = 'pulse 0.5s ease-in-out';
 }, 10);

 console.log(`Playing sound for letter: ${letter}`);
 } else {
 alert(`Letter: ${letter}`);
 }

 // Mark as completed
 this.markLetterCompleted(letter);
 }

 markLetterCompleted(letter) {
 if (this.completedLetters.has(letter)) return;

 this.completedLetters.add(letter);

 // Update UI
 const btn = document.querySelector(`.letter-btn[data-letter="${letter}"]`);
 if (btn) {
 btn.classList.add('completed');
 }

 // Update progress
 this.updateProgress();

 // Save progress
 this.saveProgress();

 console.log(`Letter ${letter} marked as completed`);
 }

 updateProgress() {
 const badge = document.getElementById('progress-badge');
 const count = this.completedLetters.size;
 badge.textContent = `Progress: ${count}/26`;

 // Celebrate completion
 if (count === 26) {
 setTimeout(() => {
 alert('&#127881; Amazing! You\'ve learned all the letters! You\'re a star! &#11088;');
 }, 500);
 }
 }

 saveProgress() {
 const data = {
 completed: Array.from(this.completedLetters),
 currentLetter: this.currentLetter,
 lastUpdated: new Date().toISOString()
 };

 localStorage.setItem('abc-learning-progress', JSON.stringify(data));
 console.log('Progress saved');
 }

 loadProgress() {
 const saved = localStorage.getItem('abc-learning-progress');
 if (!saved) return;

 try {
 const data = JSON.parse(saved);
 this.completedLetters = new Set(data.completed || []);
 this.currentLetter = data.currentLetter || 'A';

 // Update UI
 this.completedLetters.forEach(letter => {
 const btn = document.querySelector(`.letter-btn[data-letter="${letter}"]`);
 if (btn) {
 btn.classList.add('completed');
 }
 });

 this.updateProgress();
 console.log('Progress loaded');
 } catch (error) {
 console.error('Failed to load progress:', error);
 }
 }
 }

 // Add pulse animation
 const style = document.createElement('style');
 style.textContent = `
 @keyframes pulse {
 0%, 100% { transform: scale(1); }
 50% { transform: scale(1.05); }
 }
 `;
 document.head.appendChild(style);

 // Initialize when page loads
 window.addEventListener('load', () => {
 new ABCLearning();
 });
 </script>
