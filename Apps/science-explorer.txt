 init() {
 this.renderTopics();
 this.updateStats();

 analytics.logEvent('page_view', {
 page_title: 'Science Explorer',
 page_location: window.location.href
 });
 }

 renderTopics() {
 const grid = document.getElementById('topics-grid');
 grid.innerHTML = '';

 Object.entries(this.topics).forEach(([key, topic]) => {
 const card = document.createElement('div');
 card.className = 'topic-card';
 const factCount = topic.facts.length;

 card.innerHTML = `
 <div class="icon">${topic.icon}</div>
 <div class="title">${topic.title}</div>
 <div class="count">${factCount} facts</div>
 `;

 card.addEventListener('click', () => this.showTopic(key, topic));
 grid.appendChild(card);
 });
 }

 showTopic(topicKey, topic) {
 this.stats.topicsExplored.add(topicKey);
 this.updateStats();
 this.saveProgress();

 const contentArea = document.getElementById('content-area');
 contentArea.className = 'content-area active';

 let html = `<h2 style="text-align: center; color: #2D3748; margin-bottom: 30px;">${topic.icon} ${topic.title}</h2>`;

 // Add facts
 topic.facts.forEach(fact => {
 html += `
 <div class="fact-card">
 <h3>${fact.emoji} ${fact.title}</h3>
 <p>${fact.text}</p>
 </div>
 `;
 });

 // Add quiz
 if (topic.quiz && topic.quiz.length > 0) {
 html += `<div class="quiz-container"><h3 style="color: #2D3748; margin-bottom: 20px;">Test Your Knowledge!</h3>`;

 topic.quiz.forEach((q, qIndex) => {
 html += `
 <div class="quiz-question">
 <h4>${q.question}</h4>
 <div class="quiz-options">
 `;

 q.options.forEach((option, oIndex) => {
 html += `
 <div class="quiz-option" onclick="scienceExplorer.checkAnswer(${qIndex}, ${oIndex}, ${q.correct}, this)">
 ${option}
 </div>
 `;
 });

 html += `
 </div>
 </div>
 `;
 });

 html += `</div>`;
 }

 html += `
 <div class="back-to-topics">
 <button onclick="scienceExplorer.backToTopics()">&larr; Back to Topics</button>
 </div>
 `;

 contentArea.innerHTML = html;

 analytics.logEvent('select_content', {
 content_type: 'science_topic',
 item_id: topicKey
 });
 }

 checkAnswer(questionIndex, selectedIndex, correctIndex, element) {
 this.stats.quizTotal++;

 const allOptions = element.parentElement.querySelectorAll('.quiz-option');
 allOptions.forEach(opt => opt.style.pointerEvents = 'none');

 if (selectedIndex === correctIndex) {
 element.classList.add('correct');
 this.stats.quizCorrect++;
 this.stats.factsLearned++;

 analytics.logEvent('quiz_answer', {
 result: 'correct'
 });
 } else {
 element.classList.add('wrong');
 allOptions[correctIndex].classList.add('correct');

 analytics.logEvent('quiz_answer', {
 result: 'incorrect'
 });
 }

 this.updateStats();
 this.saveProgress();
 }

 backToTopics() {
 document.getElementById('content-area').classList.remove('active');
 }

 updateStats() {
 document.getElementById('topics-explored').textContent = this.stats.topicsExplored.size;
 document.getElementById('facts-learned').textContent = this.stats.factsLearned;

 const score = this.stats.quizTotal > 0
 ? Math.round((this.stats.quizCorrect / this.stats.quizTotal) * 100)
 : 0;
 document.getElementById('quiz-score').textContent = score + '%';
 }

 saveProgress() {
 const data = {
 topicsExplored: Array.from(this.stats.topicsExplored),
 factsLearned: this.stats.factsLearned,
 quizCorrect: this.stats.quizCorrect,
 quizTotal: this.stats.quizTotal
 };
 localStorage.setItem('science-explorer-progress', JSON.stringify(data));
 }

 loadProgress() {
 const saved = localStorage.getItem('science-explorer-progress');
 if (saved) {
 const data = JSON.parse(saved);
 this.stats.topicsExplored = new Set(data.topicsExplored || []);
 this.stats.factsLearned = data.factsLearned || 0;
 this.stats.quizCorrect = data.quizCorrect || 0;
 this.stats.quizTotal = data.quizTotal || 0;
 }
 }
